#  PhishGuard AI — Python Dependencies

#
#  HOW TO USE:
#    After activating your virtual environment, run:
#      pip install -r requirements.txt
#
#  WHY a requirements.txt?
#    When you share this project, teammates need to know which
#    packages to install. This file is that recipe list.
#    The "==" pins the EXACT version — so everyone gets
#    identical behaviour. Without pinning, "pip install flask"
#    today and tomorrow could give different versions.
#
#  SETUP ORDER (run these once, inside /backend):
#    1. python3 -m venv .venv        <- create isolated environment
#    2. source .venv/bin/activate    <- Mac/Linux
#       .venv\Scripts\activate       <- Windows
#    3. pip install -r requirements.txt
#
#  You'll know venv is active when your prompt shows (.venv):
#    (.venv) shwn@shwn-PC:~/code/phishguard$
#
#  DOCKER NOTE:
#    Inside Docker, pip install runs during the image build.
#    No .venv needed — the container IS the isolated environment.
#    The Dockerfile copies this file and runs pip install
#    before copying application code (layer caching optimization).
# =============================================================


# -- WEB FRAMEWORK ---------------------------------------------
# Flask     : core web framework. Turns Python functions into
#             HTTP endpoints a browser can call.
#             A "route" maps a URL to a Python function:
#               @app.route('/api/detect', methods=['POST'])
#               def detect(): ...
#
# Flask-CORS: browsers enforce the Same-Origin Policy — JS on
#             localhost:5500 is blocked from calling an API on
#             localhost:5000 by default. CORS adds response
#             headers like 'Access-Control-Allow-Origin: *'
#             that tell the browser "cross-origin is allowed here".
#             In production, restrict this to your actual domain.
Flask==3.0.3
Flask-CORS==4.0.1


# -- DATABASE --------------------------------------------------
# Flask-SQLAlchemy: ORM (Object Relational Mapper) — lets you
#             write Python classes instead of raw SQL.
#             A class = a table. An instance = a row.
#             Supports SQLite (dev), PostgreSQL (prod), MySQL, etc.
#             Switching databases = change one config URL.
#
# Flask-Migrate   : tracks schema changes over time — like Git
#             for your database. When you add a column:
#               flask db migrate -m "add gmail_connected to users"
#               flask db upgrade
#             This generates a versioned migration file and applies
#             it without losing existing data. Critical for
#             production where you cannot just delete the DB.
Flask-SQLAlchemy==3.1.1
Flask-Migrate==4.0.7


# -- AUTHENTICATION --------------------------------------------
# PyJWT  : creates and verifies JSON Web Tokens (JWTs).
#          Flow:
#            1. User logs in with username + password
#            2. Server verifies credentials, creates a signed token:
#               {"user_id": 1, "exp": <1hr from now>}
#            3. Client stores token, sends it with every request
#            4. Server verifies signature — no DB lookup needed
#          The token is signed with SECRET_KEY using HS256 algorithm.
#          If anyone tampers with the payload, the signature breaks.
#
# bcrypt : one-way password hashing algorithm.
#          NEVER store plain-text passwords. Ever.
#          bcrypt(password + salt) -> hash stored in DB.
#          On login: bcrypt.check_password_hash(stored, attempt)
#          Why bcrypt over MD5/SHA256?
#            bcrypt is intentionally slow (configurable cost factor).
#            MD5 can be brute-forced at 10 billion attempts/second.
#            bcrypt: ~100 attempts/second. Makes brute force impractical.
PyJWT==2.8.0
bcrypt==4.1.3

#                Components used in PhishGuard:
#                  TfidfVectorizer         -- converts email text to numbers
#                  RandomForestClassifier  -- the detection model
#                  train_test_split        -- splits data for evaluation
#                  accuracy_score          -- measures model performance
#
# numpy        : fast numeric array library. scikit-learn requires
#                it. All ML data is represented as numpy arrays.
#                TF-IDF output is a numpy matrix where each row = one
#                email, each column = one word's weighted score.
#
# pandas       : spreadsheet-style data handling. Used for loading
#                training data from CSV files, cleaning data, and
#                splitting into features (X) and labels (y).
#
# joblib       : serialises Python objects to disk efficiently.
#                Used to save/load the trained model:
#                  joblib.dump(model, 'model.pkl')   <- after training
#                  joblib.load('model.pkl')           <- on server start
#                Without this, we would retrain on every server restart.
scikit-learn==1.4.2
numpy==1.26.4
pandas==2.2.2
joblib==1.4.2

#                stopword removal ("the", "a", "is" add no signal),
#                stemming (running -> run, phishing -> phish).
#                Requires downloading data on first run:

#                Handles multipart emails (text + HTML + attachments),
#                decodes quoted-printable and base64 payloads,
#                extracts headers (From, Subject, Date, Received).
#                Core of the .eml upload feature (v0.2.0).

# -- PRODUCTION SERVER -----------------------------------------
# gunicorn     : production-grade WSGI server.
#                Flask's built-in server is single-threaded and
#                explicitly warns "do not use in production".
#                Gunicorn spawns multiple worker processes:
#                  --workers 4   -> 4 parallel processes
#                  --threads 2   -> 2 threads per worker
#                = 8 requests handled simultaneously.
#                Nginx sits in front of Gunicorn as a reverse proxy,
#                handling SSL, static files, and slow clients.
gunicorn==22.0.0


# -- POSTGRESQL ADAPTER ----------------------------------------
# psycopg2-binary : the bridge between SQLAlchemy and PostgreSQL.
#                SQLite limitations in production:
#                  - One writer at a time (blocks concurrent scans)
#                  - Cannot be on a separate server
#                  - Poor performance beyond ~100,000 rows
#                PostgreSQL solves all of these.
#                The -binary variant bundles the C library — no need
#                to install libpq-dev separately on the host.
#                The Dockerfile installs libpq-dev explicitly anyway
#                for maximum reliability across environments.
psycopg2-binary==2.9.9


# -- REDIS CLIENT ----------------------------------------------
# redis        : Python client for the Redis in-memory data store.
#                Used by:
#                  Flask-Limiter -> rate limit counters per IP
#                  VT cache      -> URL reputation results (24hr TTL)
#                Why Redis over memory://?
#                  memory:// is per-process. With 4 Gunicorn workers,
#                  each has its own counter — user gets 4x the limit.
#                  Redis is shared: one counter across all workers.
#                  Redis also persists across restarts — memory:// resets.
redis==5.0.6
